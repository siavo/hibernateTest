CREATE TABLE company(
                        id SERIAL PRIMARY KEY ,
                        name VARCHAR(32) unique
);

CREATE TABLE users
(
    id BIGSERIAL PRIMARY KEY ,
    username VARCHAR(128) unique ,
    firstname VARCHAR(128) ,
    lastname VARCHAR(128),
    birth_date DATE,
    role VARCHAR(32),
    company_id INT REFERENCES company(id)
);

DROP TABLE users;
DROP TABLE company;
